def plot_hist_recons(recon_errors, labels_binary, title="Reconstruction error distribution"):
    plt.figure(figsize=(8,4))
    plt.hist(recon_errors[labels_binary==0], bins=80, alpha=0.6, label='BENIGN')
    plt.hist(recon_errors[labels_binary==1], bins=80, alpha=0.6, label='ATTACK')
    plt.legend()
    plt.title(title)
    plt.xlabel("Reconstruction error (MSE)")
    plt.ylabel("Count")
    plt.show()

def plot_roc_curve(y_true, scores, title="ROC - Anomaly Detection"):
    fpr, tpr, _ = roc_curve(y_true, scores)
    auc = roc_auc_score(y_true, scores)
    plt.figure(figsize=(6,5))
    plt.plot(fpr, tpr, label=f"AUC = {auc:.4f}")
    plt.plot([0,1],[0,1],'--', color='gray')
    plt.xlabel("False Positive Rate")
    plt.ylabel("True Positive Rate")
    plt.title(title)
    plt.legend()
    plt.grid(True)
    plt.show()
    return auc

input_dim = X_train.shape[1]
latent_dim = 8

# Encoder
inputs = layers.Input(shape=(input_dim,), name="encoder_input")
h = layers.Dense(128, activation='relu')(inputs)
h = layers.Dense(64, activation='relu')(h)
z_mean = layers.Dense(latent_dim, name='z_mean')(h)
z_log_var = layers.Dense(latent_dim, name='z_log_var')(h)

def sampling(args):
    z_mean, z_log_var = args
    epsilon = K.random_normal(shape=(K.shape(z_mean)[0], latent_dim), mean=0., stddev=1.0)
    return z_mean + K.exp(0.5 * z_log_var) * epsilon

z = layers.Lambda(sampling, output_shape=(latent_dim,), name='z')([z_mean, z_log_var])

# Decoder
latent_inputs = layers.Input(shape=(latent_dim,), name='z_sampling')
x = layers.Dense(64, activation='relu')(latent_inputs)
x = layers.Dense(128, activation='relu')(x)
outputs = layers.Dense(input_dim, activation=None)(x)  # linear output for scaled continuous features

encoder = Model(inputs, [z_mean, z_log_var, z], name='encoder')
decoder = Model(latent_inputs, outputs, name='decoder')

# VAE model
z_mean_, z_log_var_, z_ = encoder(inputs)
reconstructions = decoder(z_)
vae = Model(inputs, reconstructions, name='vae')

# Loss: reconstruction (MSE) + KL
reconstruction_loss = tf.reduce_mean(tf.reduce_sum(tf.square(inputs - reconstructions), axis=1))
kl_loss = -0.5 * tf.reduce_mean(tf.reduce_sum(1 + z_log_var_ - tf.square(z_mean_) - tf.exp(z_log_var_), axis=1))
vae.add_loss(reconstruction_loss + 1e-3 * kl_loss)  # small weight on KL
vae.compile(optimizer=tf.keras.optimizers.Adam(1e-3))
vae.summary()

# Build generator
def build_generator(latent_dim, output_dim):
    inp = layers.Input(shape=(latent_dim,))
    x = layers.Dense(64, activation='relu')(inp)
    x = layers.Dense(128, activation='relu')(x)
    out = layers.Dense(output_dim, activation=None)(x)
    return Model(inp, out, name='generator')

# Build discriminator
def build_discriminator(input_dim):
    inp = layers.Input(shape=(input_dim,))
    x = layers.Dense(128, activation='relu')(inp)
    x = layers.Dense(64, activation='relu')(x)
    out = layers.Dense(1, activation='sigmoid')(x)
    return Model(inp, out, name='discriminator')

generator = build_generator(latent_g, input_dim)
discriminator = build_discriminator(input_dim)
discriminator.compile(optimizer=tf.keras.optimizers.Adam(1e-4), loss='binary_crossentropy', metrics=['accuracy'])

# Adversarial (stacked) model
discriminator.trainable = False
z_in = layers.Input(shape=(latent_g,))
gen_out = generator(z_in)
disc_out = discriminator(gen_out)
gan = Model(z_in, disc_out)
gan.compile(optimizer=tf.keras.optimizers.Adam(1e-4), loss='binary_crossentropy')
gan.summary()

def print_minority_metrics(y_true, y_pred, label):
    mask_true = (y_true == label)
    mask_pred = (y_pred == label)
    tp = np.sum(mask_true & mask_pred)
    fn = np.sum(mask_true & (~mask_pred))
    fp = np.sum((~mask_true) & mask_pred)
    print(f"Label {label} — TP: {tp}, FN: {fn}, FP: {fp}, Support (true): {np.sum(mask_true)}")

def generate_for_label(label_encoded, n=50, prompt_template=None):
    label_text = label_names.get(int(label_encoded), str(label_encoded))
    samples = []
    
    rows_label = X_tab[np.array(y) == label_encoded]
    rows_label = rows_label.sample(min(len(rows_label), 200), random_state=SEED) if len(rows_label)>0 else X_tab.sample(50, random_state=SEED)
    for i, (_, row) in enumerate(rows_label.iterrows()):
        base_prompt = prompt_template or row_to_prompt(row, label_encoded)
        out = gen(base_prompt, max_length=base_prompt.count(" ")+60, num_return_sequences=1, do_sample=True, top_p=0.9, temperature=0.8)
        text = out[0]["generated_text"]
        
        if "|| DESCRIPTION:" in text:
            text = text.split("|| DESCRIPTION:")[1].strip()
        samples.append(text)
        if len(samples) >= n:
            break
    return samples
